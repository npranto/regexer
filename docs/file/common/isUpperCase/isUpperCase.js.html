    common/isUpperCase/isUpperCase.js

    import { isNull, isUndefined, isString } from './../../utils';

const isUpperCase = str => (
  !isUndefined(str) &&
  !isNull(str) &&
  isString(str) &&
  /^[^a-z]+$/.test(str)
);

export default isUpperCase;
